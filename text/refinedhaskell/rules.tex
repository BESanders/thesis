$$
\inference{
	\undechastype{\Gamma}{e}{\tau'} &&
	\undecissubtype{\Gamma}{\tau'}{\tau} &&
	\undeciswellformed{\Gamma}{\tau} &&
}{
	\undechastype{\Gamma}{e}{\tau}
}[\rtsub]

$$
\inference{
	\undechastype{\Gamma, x\colon\tau_x}{e}{\tau} &&

}[\rtfun]

$$
\inference{
	\undechastype{\Gamma}{e_1}{(\tfunref{x}{\tau_{x}}{\tau}{v}{e_v})} &&
